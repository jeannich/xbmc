#Official doc page:
# https://github.com/xbmc/xbmc/blob/master/docs/README.macOS.md#4-configure-and-build-tools-and-dependencies


#Enter project folder:
kodi_src_root_folder="$HOME/github.com/xbmc_me"
cd $kodi_src_root_folder


# compile deps (done once):
cd tools/depends
./bootstrap
./configure --host=aarch64-apple-darwin --with-platform=macos --with-target-cxxflags=-std=c++20
make -j$(getconf _NPROCESSORS_ONLN)



# Generate XCode build files (done once):
mkdir -p $kodi_src_root_folder/build
make -C $kodi_src_root_folder/tools/depends/target/cmakebuildsys BUILD_DIR=$kodi_src_root_folder/build GEN=Xcode



# build command (run at each code change)
cd $kodi_src_root_folder/build
xcodebuild -config "Debug" -jobs 20
